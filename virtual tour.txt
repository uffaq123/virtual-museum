const tourScenes = [
  {
    image: "placeholder1.jpg", // Replace with actual image URL
    alt: "Museum Entrance",
    description: "Welcome to the grand museum. You see the main entrance with tall pillars and a large wooden door."
  },
  {
    image: "placeholder2.jpg", // Replace with actual image URL
    alt: "Main Hall",
    description: "You are now inside the main hall, with high ceilings and beautiful chandeliers hanging from above."
  },
  {
    image: "placeholder3.jpg", // Replace with actual image URL
    alt: "Exhibit Room",
    description: "This room contains ancient artifacts from different civilizations, with glass displays lining the walls."
  }
];

let currentIndex = 0;
let utterance = null;
const totalScenes = tourScenes.length;

// Get DOM elements
const imageElement = document.getElementById('locationImage');
const startBtn = document.getElementById('startTour');
const nextBtn = document.getElementById('next');
const prevBtn = document.getElementById('prev');
const repeatBtn = document.getElementById('repeat');
const pauseBtn = document.getElementById('pauseTour');
const resumeBtn = document.getElementById('resumeTour');
const mapContainer = document.getElementById('mapContainer');

// Generate map buttons dynamically
tourScenes.forEach((scene, index) => {
  const btn = document.createElement('button');
  btn.innerText = `Scene ${index + 1}`;
  btn.setAttribute('aria-label', `Go to scene ${index + 1}`);
  btn.style.padding = '8px 12px';
  btn.style.borderRadius = '4px';
  btn.onclick = () => {
    currentIndex = index;
    loadScene(currentIndex);
  };
  mapContainer.appendChild(btn);
});

// Function to speak text
function speak(text) {
  window.speechSynthesis.cancel();
  utterance = new SpeechSynthesisUtterance(text);
  window.speechSynthesis.speak(utterance);
}

// Function to load a scene
function loadScene(index) {
  const scene = tourScenes[index];
  imageElement.src = scene.image;
  imageElement.alt = scene.alt;
  // Speak description
  speak(scene.description);
}

// Initialize the tour
function startTour() {
  currentIndex = 0;
  loadScene(currentIndex);
}

// Navigation functions
function nextScene() {
  if (currentIndex < totalScenes - 1) {
    currentIndex++;
    loadScene(currentIndex);
  } else {
    window.speechSynthesis.cancel();
    alert("You have reached the end of the tour.");
  }
}

function prevScene() {
  if (currentIndex > 0) {
    currentIndex--;
    loadScene(currentIndex);
  } else {
    alert("This is the first scene.");
  }
}

function repeatScene() {
  if (utterance && window.speechSynthesis.speaking) {
    window.speechSynthesis.cancel();
  }
  speak(tourScenes[currentIndex].description);
}

// Event Listeners
startBtn.onclick = startTour;
nextBtn.onclick = nextScene;
prevBtn.onclick = prevScene;
repeatBtn.onclick = repeatScene;

pauseBtn.onclick = () => {
  window.speechSynthesis.pause();
};
resumeBtn.onclick = () => {
  window.speechSynthesis.resume();
};

// Keyboard controls for accessibility
document.addEventListener('keydown', (e) => {
  switch (e.key) {
    case 'ArrowRight':
      nextScene();
      break;
    case 'ArrowLeft':
      prevScene();
      break;
    case 'Enter':
    case ' ':
      repeatScene();
      break;
    case 'p':
      window.speechSynthesis.pause();
      break;
    case 'r':
      window.speechSynthesis.resume();
      break;
  }
});
